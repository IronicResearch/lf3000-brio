46 // Type:
47 // tCharEncoding
48 //
49 // Definition:
50 // Character encodings extend character and string support beyond
51 // Clib ASCII to include Windows codepages, ISO-8859 parts, and Unicode
52 // among other common forms & standards.
53 //
54 // tCharEncoding stores a character encoding value, comprised of three elements:
55 //
56 // MSBit: [4 bit domain][8 bits reserved][10 bit group][10 bit tag] :LSBit
57 //
58 // - the domain, which segments system, product and application encodings
59 // - the group, which specifies the group to which the encoding belongs
60 // - the tag - a key number that identifies the specific sub-encoding
61 // within its group
62 //
63 // Application code may use any product or System-defined types, but
64 // can only define new application types -- not product or System
65 // types. Similarly, product code may use any System-defined types,
66 // but can only define new product types -- not application or System types.
67 //------------------------------------------------------------------------------

167 //==============================================================================
168 // Type:
169 // tWChar
170 //
171 // Definition:
172 // This 4-byte type is big enough to hold any single character regardless of
173 // the particular text encoding. It provides international & multi-byte
174 // character support in lieu of the C-library wchar_t type. We don't
175 // use wchar_t or related Clib string functionality because the Clib
176 // "wide character" definition can vary in size from 8 to 32 bits depending
177 // on the compiler and library settings, and wchar_t string processing
178 // typically depends on a single text encoding scheme.
179 //
180 // tWChars are related to tCharEncoding-encoded strings in that any single
181 // character, when represented outside of an encoded string, is of type tWChar.
182 // However, within the encoded string, individual character storage size &
183 // format is entirely determined by the character encoding.
184 //
185 // tWChars themselves provide no information about their character set,
186 // format or byte-size. No assumptions can be made about any of these attributes.
187 // tWChar objects must be paired with a tCharEncoding type that defines the
188 // format of the character.
189 //==============================================================================
190 
191 typedef U32 tWChar; // storage type large enough for a single character of any encoding
